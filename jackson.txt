    function validArea() {
        var min = $("#MinArea").val();

        var error =$("#MaxAreaError");
        if(min&&max)
        {
            if (parseInt(min) > parseInt(max)) {
                error.html('<span for="MaxArea" class="">最大面积>=最小面积</span>');
                error.addClass('field-validation-error').removeClass('field-validation-valid');
                return ;
            }
            if (parseInt(min) * 1.5 < parseInt(max)) {
                error.html('<span for="MaxArea" class="">最大面积<=最小面积*150%</span>');
                error.addClass('field-validation-error').removeClass('field-validation-valid');
                return ;
